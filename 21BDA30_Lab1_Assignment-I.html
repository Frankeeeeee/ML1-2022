class b_a:
    def __init__(self):
        self.balance=0
        print("Bank Account details")
    def deposit(self):
        amt=float(input("enter amt to be deposited"))
        self.balance+=amt
        print("Amount deposited",amt)
        
    def withdraw(self):
        amt=float(input("enter amt to be deposited"))
        if self.balance>=amt:
            self.balance-=amt
            print("Amt withdrawn",amt)
        else:
            print("Sorry, no balance")
            
    def balance1(self):
        print("Net balance",self.balance)
